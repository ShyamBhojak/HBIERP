service cloud.firestore {
  match /databases/{database}/documents {
    
    // Define your authorized users in a constant
    function isAuthorized() {
      return request.auth != null && 
             (request.auth.token.email == "inquiry@hbinstitute.co.in" || 
              request.auth.token.email == "shyambhojak25@gmail.com" ||
              request.auth.token.email == "shyambhojak19@gmail.com" ||
              request.auth.token.email == "hdbhojak@gmail.com" ||
              request.auth.token.email == "gumnani.komal@gmail.com");
    }

    match /artifacts/{appId}/public/data/{document=**} {
      // Only allow operations if the user is in the authorized email list
      allow read, write: if isAuthorized();
    }
    // In Firebase Rules
  	match /artifacts/{appId}/public/data/students/{studentId} {
    // Admins can read/write everything
    allow read, write: if isAuthorized();

    // Students can ONLY read their own document (based on email match)
    allow read: if request.auth.token.email == resource.data.email;
  }
    
    
  }
